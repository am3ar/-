-- جدول المقالات CREATE TABLE articles ( id INT AUTO_INCREMENT PRIMARY KEY, title VARCHAR(255) NOT NULL, content TEXT NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- جدول الفتاوى CREATE TABLE fatwas ( id INT AUTO_INCREMENT PRIMARY KEY, title VARCHAR(255) NOT NULL, content TEXT NOT NULL, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- جدول الدروس CREATE TABLE lessons ( id INT AUTO_INCREMENT PRIMARY KEY, title VARCHAR(255) NOT NULL, description TEXT, video_url VARCHAR(255), created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- جدول القرآن الكريم CREATE TABLE quran ( id INT AUTO_INCREMENT PRIMARY KEY, surah_name VARCHAR(255) NOT NULL, ayah_number INT NOT NULL, verse TEXT NOT NULL, interpretation TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP );
